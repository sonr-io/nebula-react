# Button
Sonr SSO button component.

# Properties
```js
{
    id?: string;
    type?: 'submit' | 'reset' | 'button' | undefined;           // identifier
    size?: 'lg' | 'md' | 'sm';                                  // to set the height of the button, based on the Tailwind.css height h-10, h-8 and h-6 (default 'md')
    skin?: 'primary' | 'secondary' | 'subtle' | 'transparent';  // to set the overal display of the button in terms of colors (default 'primary')
    disabled?: boolean;                                         // if the button is disabled
    styling?: string;                                           // to expand the styling with your custom css classes
    label: string;                                              // label for the button
    iconName?: IconNames;                                       // icon to be displayed, the color will be subjected to the 'skin' propperty, and size will follow the 'size' propperty
    iconType?: 'filled' | 'outline' | 'duotone';                // to set the icon variation for the name you provided (default 'filled' but only applied when with 'iconName')
    onClick?: MouseEventHandler | undefined;                    // callback function
}
```
